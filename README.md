# Dependencias
    coreutils
    cmake
    build-essential
    qt5-default
    libsdl2-dev
    libsdl2-image-dev
    libsdl2-ttf-dev
    liblua5.3-dev
    libyaml-cpp-dev
    libsdl2-mixer-dev

# Instalacion de dependencias
    sh dependencies.sh

# Para compilar
    mkdir build
    cd build 
    cmake ..
    make

# Para Instalar
    sh install.sh
    
# Para Desisntalar
    sh uninstall.sh
